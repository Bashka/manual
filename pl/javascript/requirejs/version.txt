====== Мультиверсионность ======
Если в проекте используется несколько модулей разных версий, возможно создать несколько загрузчиков с различными конфигурациями:

<code javascript>
var require1 = requirejs.config({
  context: 'ver1',
  baseUrl: 'scripts/vendor/ver1'
});

require1([...], function(...){
  ...
});

var require2 = requirejs.config({
  context: 'ver2',
  baseUrl: 'scripts/vendor/ver2'
});

require2([...], function(...){
  ...
});
</code>
